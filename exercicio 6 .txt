# 6) Escreva um algoritmo para ler as dimensões de um retângulo (base e altura), calcular e escrever a área do retângulo.
print("Escreva a base do retangulo")
base = float(input())

print("Escreva a altura do retangulo")
altura = float(input())

area = base * altura

print("A area do retangulo é : " + str(area))
